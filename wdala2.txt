<script>
    const star = document.getElementById('star');
    let angle = 0;

    function rotate() {
        angle += 1;  // Rotate by 1 degree each frame
        star.style.transform = `rotate(${angle}deg)`;
        requestAnimationFrame(rotate);
    }

    rotate();
</script>
